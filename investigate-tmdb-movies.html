# This function will be use throughout this notebook, therefore, we are declaring it in this section
#The purpose of this function is to count the values in categories where elements are separated by the pipe character (|)
def CountColumnsWithPipe(column):
    # Separate elements by pipe
    d = movies_df[column].str.cat(sep = '|')
    # make array from elements separateed by pipe
    d = pd.Series(d.split('|'))
    # count the total vlues
    total = d.value_counts(ascending = False) 
    return total
